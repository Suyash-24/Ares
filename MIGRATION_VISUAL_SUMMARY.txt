🔄 API CHANGES AT A GLANCE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  Manager Access:
    OLD: client.manager.players.get(guildId)
    NEW: client.queue.get(guildId)

  Add Track:
    OLD: player.queue.add(track)
    NEW: queue.addTrack(track)

  Play:
    OLD: player.play()
    NEW: await queue.play()

  Volume:
    OLD: player.setVolume(50)           [range: 0-100]
    NEW: await queue.setVolume(50)      [range: 1-1000]

  Track Property:
    OLD: track.title, track.author
    NEW: track.info.title, track.info.author
